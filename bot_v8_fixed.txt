def is_admin(user_id: int) -> bool:
    if not ADMIN_IDS:
        return True  # Hiç admin tanımlı değilse herkese izin ver
    return user_id in ADMIN_IDS

def admin_only(func):
    @functools.wraps(func)
    async def wrapper(event):
        if not is_admin(event.sender_id):
            await event.reply("🚫 Bu komutu kullanma yetkiniz yok.")
            return
        await func(event)
    return wrapper

def log(msg, level="INFO"):
    ts = datetime.now().strftime("%H:%M:%S")
    print(f"[{ts}] [{level}] {msg}", flush=True)
    if level == "ERROR":
        try:
            entry = f"[{datetime.now().strftime('%d.%m.%Y %H:%M:%S')}] {msg}\n"
            lines = []
            if os.path.exists(ERROR_LOG_FILE):
                with open(ERROR_LOG_FILE, 'r', encoding='utf-8') as f:
                    lines = f.readlines()
            lines.append(entry)
            if len(lines) > MAX_ERROR_LINES:
                lines = lines[-MAX_ERROR_LINES:]
            with open(ERROR_LOG_FILE, 'w', encoding='utf-8') as f:
                f.writelines(lines)
        except Exception:
            pass

def log_error(msg):
    log(msg, level="ERROR")

def get_next_game(n):

def clean_text(t):
    t = re.sub(r'[\uFE00-\uFE0F]', '', t)      # emoji varyasyon seçicileri
    t = re.sub(r'\*\*([^*]*)\*\*', r'\1', t)   # **bold** → düz metin
    t = re.sub(r'__([^_]*)__', r'\1', t)        # __underline__ → düz metin
    t = re.sub(r'\[([^\]]*)\]\([^)]*\)', r'\1', t)  # [metin](url) → metin
    t = re.sub(r'\s+', ' ', t)
    return t.strip()

def split_cards(card_str):
    if not card_str:
        return []
    return re.findall(r'(\d+|J|Q|K|A)\s*([♣♦♥♠])', card_str)

def baccarat_card_value(rank):
    return BACCARAT_VALUES.get(rank, 0)

def baccarat_score(cards):
    """cards: list of (rank, suit) tuples — ilk N kartın Bakara skorunu hesaplar"""
    return sum(baccarat_card_value(r) for r, _ in cards) % 10

def extract_full_game_info(text):
    """Mesajdan oyun bilgilerini çıkarır — yeni strateji için geliştirilmiş parser"""
    text = clean_text(text)
    result = {
        "game_number": None,
        "is_final": False,
        "is_tie": False,
        "c2_2": False,
        "winner_cards": [],
        "all_cards_str": "",
        "player_cards_str": "",  # Sol taraf (Oyuncu) kartları
        "winner_initial_score": None,
        "winner_drew_third": False,
        "winner_third_suit": None,
    }

    m = re.search(r'(?:#N|№)\s*(\d+)', text)
    if not m:
        return result
    result["game_number"] = int(m.group(1))

    if '⏱' not in text and ('WinWin' in text or 'Melbet' in text or '✅' in text or '#X' in text):
        result["is_final"] = True

    if '#X' in text or '🔰' in text:
        result["is_tie"] = True

    if '#C2_2' in text:
        result["c2_2"] = True

    def parse_cards(card_str):
        return re.findall(r'(10|[2-9JQKA])\s*([♣♦♥♠])', card_str)

    all_parens = re.findall(r'\(([^)]+)\)', text)
    result["all_cards_str"] = " ".join(all_parens)
    result["player_cards_str"] = all_parens[0] if all_parens else ""  # Sol taraf = Oyuncu

    m1 = re.search(r'✅\s*\d+\s*\(([^)]+)\)\s*-\s*\d+\s*\(([^)]+)\)', text)
    if m1:
        result["winner_cards"] = parse_cards(m1.group(1))
    else:
        m2 = re.search(r'\d+\s*\(([^)]+)\)\s*-\s*✅\s*\d+\s*\(([^)]+)\)', text)
        if m2:
            result["winner_cards"] = parse_cards(m2.group(2))

    if not result["winner_cards"] and result["is_final"] and not result["is_tie"] and not result["c2_2"]:
        first_line = text.split('\n')[0][:120]
        log(f"🔍 Parse edilemeyen satır: {first_line}")

    if result["winner_cards"] and len(result["winner_cards"]) >= 2:
        result["winner_initial_score"] = baccarat_score(result["winner_cards"][:2])
        if len(result["winner_cards"]) >= 3:
            result["winner_drew_third"] = True
            result["winner_third_suit"] = result["winner_cards"][2][1]

    return result

def analyze_new_strategy(game_info):
    """
    4 kural filtresi uygular, geçerse (hedef_oyun, renk) döndürür.
    Kural 1 — #C2_2: her iki taraf da 2 kartla bitmişse → sistem dışı
    Kural 2 — Beraberlik: #X veya 🔰 varsa → sistem dışı
    Kural 3 — İlk skor 0: kazananın 2 kartlık skoru 0 ise → sistem dışı
    Kural 4 — 3. kart yok: ilk skor > 0 ama kazanan 3. kart almamışsa → sistem dışı
    """
    gnum = game_info["game_number"]

    if game_info["c2_2"]:
        log(f"⛔ Kural 1 (#C2_2): #{gnum} — sistem dışı")
        return None

    if game_info["is_tie"]:
        log(f"⛔ Kural 2 (Beraberlik): #{gnum} — sistem dışı")
        return None

    if not game_info["winner_cards"]:
        log(f"⚠️ #{gnum} — kazanan kartı okunamadı, pas")
        return None

    initial_score = game_info["winner_initial_score"]
    if initial_score is None:
        return None

    if initial_score == 0:
        log(f"⛔ Kural 3 (İlk skor 0): #{gnum} — sistem dışı")
        return None

    if not game_info["winner_drew_third"]:
        log(f"⛔ Kural 4 (3. kart yok): #{gnum}, İlk Skor:{initial_score} — sistem dışı")
        return None

    target_game = gnum + initial_score
    if target_game > MAX_GAME_NUMBER:
        target_game = target_game - MAX_GAME_NUMBER

    suit = game_info["winner_third_suit"]
    log(f"✅ SİNYAL ADAYI: #{gnum} → Hedef #{target_game} | Renk:{suit} | İlk Skor:{initial_score}")
    return (target_game, suit)

SUIT_NAMES = {'♣': 'Sinek', '♦': 'Karo', '♥': 'Kupa', '♠': 'Maça'}

def step_bar(current_step, max_steps):
    filled = "🔵" * (current_step + 1)
    empty  = "⚪" * (max_steps - current_step - 1)
    labels = ["Giriş", "1. Katlama", "2. Katlama", "3. Katlama", "4. Katlama", "5. Katlama"]
    label  = labels[current_step] if current_step < len(labels) else f"{current_step}. Katlama"
    return f"{filled}{empty}  {label}"

async def check_streak_notification(bot_client):
    global bot_paused
    # Kazanç serisi bildirimi kaldırıldı — sadece AutoStop aktif
    if current_streak <= -STREAK_NOTIFY:
        if AUTOSTOP_ENABLED and abs(current_streak) >= AUTOSTOP_LIMIT:
            bot_paused = True
            await bot_client.send_message(
                TARGET_CHANNEL,
                f"🛑 Sistem Durumu: Durduruldu\n"
                f"🤖 Kayıp Serisi: {abs(current_streak)} Oyun\n"
                f"⏸️ Kasa Koruması Devreye Girdi\n"
                f"▸ Devam için: /resume"
            )
            log(f"🛑 AutoStop: {abs(current_streak)} kayıp serisi.")

# ========================= SİNYAL İŞLEMLERİ =========================
async def send_signal(bot_client, game_num, suit):
    global is_signal_active, stats, last_signal_time
    if bot_paused or is_signal_active:
        return False
    if SIGNAL_COOLDOWN > 0 and last_signal_time:
        elapsed = (datetime.now() - last_signal_time).total_seconds() / 60
        if elapsed < SIGNAL_COOLDOWN:
            log(f"⏳ Cooldown: {SIGNAL_COOLDOWN - elapsed:.1f} dk kaldı.")
            return False

    check_daily_reset()
    motivation = f"\n💬 {random.choice(motivation_messages)}" if current_streak < 0 else ""
    text = (
        f"🤖 Oyun No: #{game_num}\n"
        f"🎯 Hedef Süit: {suit}  {SUIT_NAMES.get(suit, '')}\n"
        f"🔁 Kasa Katlama: {MAX_MARTINGALE_STEPS} Oyun"
        f"{motivation}"
    )
    try:
        if ANNOUNCE_MODE:
            sent = await bot_client.send_message(TARGET_CHANNEL, text)
            msg_id = sent.id
        else:
            msg_id = 0
            log(f"🔇 Sessiz mod — sinyal kanala gönderilmedi: #{game_num} -> {suit}")

        pending_signals[game_num] = {
            "msg_id": msg_id,
            "suit": suit,
            "step": 0,
            "max_steps": MAX_MARTINGALE_STEPS,
            "expected_game": game_num,
            "original_game": game_num,
        }
        is_signal_active = True
        last_signal_time = datetime.now()
        suit_signal_count[suit] = suit_signal_count.get(suit, 0) + 1
        stats["total_signals"] += 1
        daily_stats["total"] += 1
        save_stats()
        log(f"🚀 SİNYAL #{game_num} -> {suit}")
        return True
    except FloodWaitError as e:
        await asyncio.sleep(e.seconds)
        return await send_signal(bot_client, game_num, suit)
    except Exception as e:
        log_error(f"Sinyal hatası: {e}")
        return False

async def update_signal_result(bot_client, game_num, won, player_cards):
    global is_signal_active, current_streak
    if game_num not in pending_signals:
        return

    info = pending_signals[game_num]
    msg_id = info["msg_id"]
    expected = info["suit"]
    step = info["step"]
    max_steps = info["max_steps"]

    original_game = info.get("original_game", game_num)
    game_range = f"#{original_game} ➜ #{game_num}" if original_game != game_num else f"#{game_num}"

    if won:
        stats["correct"] += 1
        daily_stats["correct"] += 1
        suit_stats[expected]["correct"] += 1
        current_streak = current_streak + 1 if current_streak >= 0 else 1
        add_to_history(game_num, expected, True, step)
        step_icons = ["1️⃣", "2️⃣", "3️⃣", "4️⃣", "5️⃣", "6️⃣"]
        step_icon = step_icons[step] if step < len(step_icons) else f"{step+1}."
        new_text = (
            f"✅ Sistem Güncellemesi: Kazandı!\n"
            f"🤖 Oyun No: {game_range}\n"
            f"🎉 Durum: Süit Oyuncuya Geldi {expected}\n"
            f"💰 Kasa Katlama: Başarılı ({step_icon} adımda)\n"
            f"💬 {random.choice(motivation_messages)}"
        )
        log(f"✅ #{game_num} KAZANÇ — Seri:{current_streak}")
        del pending_signals[game_num]
        is_signal_active = False
        save_stats()
        await check_streak_notification(bot_client)
    else:
        if step + 1 < max_steps:
            next_step = step + 1
            next_game = get_next_game(game_num)
            info["step"] = next_step
            info["expected_game"] = next_game
            pending_signals[next_game] = pending_signals.pop(game_num)
            log(f"🔄 #{game_num} -> Katlama {next_step}, Hedef #{next_game}")
            step_icons = ["1️⃣", "2️⃣", "3️⃣", "4️⃣", "5️⃣", "6️⃣"]
            step_icon = step_icons[next_step] if next_step < len(step_icons) else f"{next_step+1}."
            katlama_text = (
                f"🔄 Sistem Güncellemesi: Katlama\n"
                f"🤖 Yeni Oyun No: #{next_game}\n"
                f"🎯 Hedef Süit: {expected}  {SUIT_NAMES.get(expected, '')}\n"
                f"🔁 Kasa Katlama: {step_icon} adım\n"
                f"💬 {random.choice(motivation_messages)}"
            )
            if ANNOUNCE_MODE and msg_id:
                try:
                    await bot_client.edit_message(TARGET_CHANNEL, msg_id, katlama_text)
                except Exception as e:
                    log_error(f"Katlama mesaj güncellenemedi: {e}")
            return
        else:
            stats["wrong"] += 1
            daily_stats["wrong"] += 1
            suit_stats[expected]["wrong"] += 1
            current_streak = current_streak - 1 if current_streak <= 0 else -1
            add_to_history(game_num, expected, False, step)
            new_text = (
                f"❌ Sistem Güncellemesi: Kayıp\n"
                f"🤖 Oyun No: {game_range}\n"
                f"💔 Durum: Süit Gelmedi {expected}\n"
                f"📉 Kasa Katlama: Başarısız ({max_steps}/{max_steps} adım)\n"
                f"💬 {random.choice(motivation_messages)}"
            )
            log(f"❌ #{game_num} KAYIP — Seri:{current_streak}")
            del pending_signals[game_num]
            is_signal_active = False
            save_stats()
            await check_streak_notification(bot_client)

    if ANNOUNCE_MODE and msg_id:
        try:
            await bot_client.edit_message(TARGET_CHANNEL, msg_id, new_text)
        except Exception as e:
            log_error(f"Mesaj düzenleme hatası: {e}")

# ========================= ANA HANDLER =========================
async def handle_message(bot_client, event):
    global last_processed_game, game_count
    msg = event.message
    if not msg or not msg.text:
        return
    info = extract_full_game_info(msg.text)
    if info["game_number"] is None:
        return
    gnum = info["game_number"]

    # Bekleyen sinyal varsa: hedef oyun geldi mi?
    if gnum in pending_signals and info["player_cards_str"]:
        expected_suit = pending_signals[gnum]["suit"]
        extracted_suits = {s for _, s in split_cards(info["player_cards_str"])}
        won = expected_suit in extracted_suits
        if won or info["is_final"]:
            await update_signal_result(bot_client, gnum, won, info["all_cards_str"])

    # Yeni sinyal analizi (sadece final mesajlarda, her oyun bir kez)
    if info["is_final"] and not is_signal_active and not bot_paused:
        if gnum != last_processed_game:
            # Parse başarısızsa (kart okunamadı) last_processed_game güncelleme —
            # böylece MessageEdited geldiğinde aynı oyun tekrar denenebilir.
            parsed_ok = bool(info["winner_cards"]) or info["c2_2"] or info["is_tie"]
            if parsed_ok:
                last_processed_game = gnum
            game_count += 1
            log(f"📊 #{gnum} işlendi (toplam: {game_count})")
            result = analyze_new_strategy(info)
            if result:
                target_game, suit = result
                await send_signal(bot_client, target_game, suit)

    @client.on(events.NewMessage(pattern=r'^/errors(\s+\d+)?$'))
    @admin_only
    async def errors_cmd(event):
        n = 10
        if event.pattern_match.group(1):
            n = min(int(event.pattern_match.group(1).strip()), 50)
        if not os.path.exists(ERROR_LOG_FILE):
            await event.reply("✅ Hata kaydı yok — her şey yolunda!")
            return
        with open(ERROR_LOG_FILE, 'r', encoding='utf-8') as f:
            lines = f.readlines()
        if not lines:
            await event.reply("✅ Hata kaydı boş.")
            return
        recent = lines[-n:]
        text = f"🔴 Son {len(recent)} Hata Kaydı\n{'─'*20}\n" + "".join(recent)
        if len(text) > 4000:
            text = text[-4000:]
        await event.reply(text)

    @client.on(events.NewMessage(pattern=r'^/session_start$'))

        session_data["start_total"] = stats["total_signals"]

            f"⏰ {session_data['start_time'].strftime('%H:%M:%S')}\n"
            f"📊 Başlangıç: ✅{stats['correct']} ❌{stats['wrong']}"
        )

    @client.on(events.NewMessage(pattern=r'^/session_end$'))
    @admin_only
    async def session_end_cmd(event):
        if not session_data["active"]:
            await event.reply("❌ Aktif oturum yok. /session_start ile başlatın.")
            return
        now = datetime.now()
        duration = now - session_data["start_time"]
        minutes = int(duration.total_seconds() / 60)
        ses_correct = stats["correct"] - session_data["start_correct"]
        ses_wrong = stats["wrong"] - session_data["start_wrong"]
        ses_total = stats["total_signals"] - session_data["start_total"]
        ses_rate = (ses_correct / ses_total * 100) if ses_total else 0
        session_data["active"] = False
        await event.reply(
            f"⏹️ **Oturum Bitti!**\n{'─'*18}\n"
            f"⏰ Süre: {minutes} dakika\n"
            f"📨 Sinyal: {ses_total}\n"
            f"✅ Kazanç: {ses_correct}\n"
            f"❌ Kayıp: {ses_wrong}\n"
            f"🎯 Oran: %{ses_rate:.1f}"
        )
